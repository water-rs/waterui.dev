use waterui::prelude::*;

pub fn counter() -> impl View {
    let count = Binding::int(0);
    let doubled = count.map(|n| n * 2);

    vstack((
        text!("Count: {count}"),
        text!("Doubled: {doubled}")
            .font_size(20),
        hstack((
            button("Increment")
                .action_with(&count, |c| c.increment(1)),
            button("Reset")
                .action_with(&count, |c| c.set(0)),
        ))
    ))
}
